* Emacs kit Dictem

*** Load the module
#+begin_src emacs-lisp
  (add-to-list 'load-path (expand-file-name 
                           "dictem" (expand-file-name
			                "elisp" dotfiles-dir)))
  (require 'dictem)
#+end_src

*** Configure server
#+begin_src emacs-lisp
(setq dictem-server "localhost")
(setq dictem-port   "2628")
(dictem-initialize)
#+end_src
